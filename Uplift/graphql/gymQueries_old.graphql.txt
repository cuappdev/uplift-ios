
query AllGyms {
    gyms {
        name
        id
        imageUrl
        description
        popular
        times {
            day
            startTime
            endTime
        }
        facilities {
            name
            details {
                detailsType
                equipment {
                  equipmentType
                  name
                  quantity
                  workoutType
                }
                items
                prices
                subFacilityNames
                times {
                  day
                  timeRanges {
                    startTime
                    endTime
                    specialHours
                    restrictions
                  }
                }
            }
        }
    }
}
